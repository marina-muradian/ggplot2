Principais Gráficos para Variáveis Qualitativas

Gráfico de Colunas

Deixando a base de dados em formato tidy e aplicando algumas transformações nos dados:

gapminder %>% 
  filter(year == 2007) %>% 
  group_by(continent) %>% 
  summarise(n = n()) %>% 
  mutate(prop = round(n/sum(n)*100,2))
# A tibble: 5 x 3
  continent     n  prop
  <fct>     <int> <dbl>
1 Africa       52 36.6 
2 Americas     25 17.6 
3 Asia         33 23.2 
4 Europe       30 21.1 
5 Oceania       2  1.41
gapminder %>% 
  filter(year == 2007) %>% 
  count(continent) %>% 
  mutate(prop = round(n/sum(n)*100,2)) %>% 
  ggplot(aes(x = reorder(continent, prop), y = prop))+
  geom_col(fill = "lightblue")+
  theme_hc()+
  labs(x = "",
       y = "Frequência Relativa (em %)",
       title = "Participação de cada continente na amostra do Gapminder (2007)",
       caption = "Fonte: Gapminder")

Gráfico de Pizza (!)

gapminder %>% 
  filter(year == 2007) %>% 
  count(continent) %>% 
  arrange(desc(continent)) %>% 
  mutate(prop = round(n/sum(n)*100,2)) %>% 
  ggplot(aes(x = "",y = prop, fill = continent))+
  geom_bar(stat = "identity", width = 1, color = "white")+
  coord_polar("y", start = 0)+
  theme_void()+
  scale_fill_brewer(palette = "Set1")+
  labs(title = "Participação de Continentes da base Gapminder (2007)",
       fill = "Continent")

Principais Gráficos para Variáveis Quantitativas

Dotplot

gapminder %>% 
  filter(year == "2007") %>% 
  select(country, gdpPercap) %>% 
  top_n(20) %>% 
  ggplot(aes(y = reorder(country, gdpPercap), x =gdpPercap))+
  geom_point(color = "darkgrey")+
  theme_hc()+
  labs(title = "Noruega lidera ranking do PIB per Capita (2007)",
       y = "",
       x = "",
       caption = "Fonte: Gapminder")
Selecting by gdpPercap

Histograma

gapminder %>% 
  filter(year == 2007) %>% 
  ggplot(aes(x = lifeExp))+
  geom_histogram(fill = "blue", col = "black")+
  theme_hc()+
  labs(title = "Expectativa de Vida em 2007",
       x = "Expectativa de Vida",
       y = "Frequência Absoluta")
`stat_bin()` using `bins = 30`. Pick better value with `binwidth`.

Histograma Alisado (Density plot)

gapminder %>% 
  filter(year %in% c(1997,2007)) %>% 
  select(lifeExp, year) %>% 
  mutate(year = factor(year)) %>% 
  ggplot(aes(x = lifeExp, color = year))+
  geom_density()+
  scale_color_manual(values = c("darkgreen","lightgreen"))+
  theme_hc()+
  labs(title = "Expectativa de Vida",
       x = "Expectativa de Vida",
       y = "Densidade",
       color = "")

Boxplot

gapminder %>% 
  filter(year %in% c(1997,2007)) %>% 
  mutate(year = factor(year)) %>% 
  ggplot(aes(x = lifeExp , y = year))+
  geom_boxplot()+
  labs(title = "Expectativa de Vida",
       x = "Expectativa de Vida",
       y = "")+
  coord_flip()+
  theme(axis.ticks.x=element_blank(),
      panel.background = element_blank(),
      axis.line.y = element_line(),
      axis.line.x = element_line())
